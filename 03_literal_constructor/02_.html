function average(...args) {
    var max = args.length;
    var sum = 0;
    var num = 0;
    for(var i=0; i<max; i++){
        var aMax = args[i].length;
        if(Array.isArray(args[i]) === true){
            for(var j=0; j<aMax; j++){
                sum += args[i][j];
                num++;
            }
        }else{
            sum += args[i];
            num++;
        }
    }
    console.log(sum/num);
}
